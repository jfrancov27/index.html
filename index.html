import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js"; import { getFirestore, collection, addDoc, getDocs, onSnapshot, deleteDoc, doc } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js"; // --- CRITICAL: YOU MUST PASTE YOUR ACTUAL CONFIG HERE --- const firebaseConfig = { apiKey: "YOUR_API_KEY", authDomain: "YOUR_PROJECT_ID.firebaseapp.com", projectId: "YOUR_PROJECT_ID", storageBucket: "YOUR_PROJECT_ID.appspot.com", messagingSenderId: "YOUR_SENDER_ID", appId: "YOUR_APP_ID" }; // Initialize Firebase const app = initializeApp(firebaseConfig); const db = getFirestore(app); const bookingsRef = collection(db, "bookings"); // Make variables accessible let selectedSeats = []; // 1. ATTACH BUTTONS MANUALLY (The most reliable way) document.addEventListener('DOMContentLoaded', () => { const btnConfirm = document.getElementById('btn-confirm'); btnConfirm.addEventListener('click', async () => { console.log("Confirm button clicked!"); // Check this in Browser Console const name = document.getElementById('user-name').value; const adults = parseInt(document.getElementById('adult-count').value) || 0; const children = parseInt(document.getElementById('child-count').value) || 0; // Validation if (!name) { alert("Please enter a name for the reservation."); return; } if (selectedSeats.length === 0) { alert("No seats selected!"); return; } try { console.log("Attempting to save to Firebase..."); // Loop through selected seats and save each for (const seat of selectedSeats) { await addDoc(bookingsRef, { seatId: seat, customerName: name, adults: adults, children: children, timestamp: new Date() }); } console.log("Save successful!"); alert(`Success! ${selectedSeats.length} seats reserved for ${name}.`); // Reset UI selectedSeats = []; document.getElementById('booking-modal').classList.add('hidden'); document.getElementById('purchase-bar').classList.add('hidden'); document.getElementById('user-name').value = ""; } catch (error) { console.error("Firebase Error:", error); alert("Database Error: Make sure your Firebase Config is correct and Firestore rules are set to Test Mode."); } }); // Reset Database Button document.getElementById('btn-reset').onclick = async () => { if(confirm("Wipe all bookings?")) { const snap = await getDocs(bookingsRef); snap.forEach(async (d) => await deleteDoc(doc(db, "bookings", d.id))); location.reload(); } }; }); // 2. REAL-TIME GRID UPDATES onSnapshot(bookingsRef, (snapshot) => { const taken = []; let revenue = 0; snapshot.forEach(doc => { const data = doc.data(); taken.push(data.seatId); revenue += (data.adults * 10) + (data.children * 5); }); renderGrid(taken); document.getElementById('report-revenue').innerText = revenue; document.getElementById('report-count').innerText = taken.length; }); function renderGrid(taken) { const grid = document.getElementById('seating-grid'); grid.innerHTML = ''; const rows = ['A', 'B', 'C', 'D', 'E']; rows.forEach(row => { for (let i = 1; i <= 8; i++) { const id = `${row}${i}`; const isOccupied = taken.includes(id); const isSelected = selectedSeats.includes(id); const seat = document.createElement('div'); seat.className = `seat w-10 h-10 rounded-md flex items-center justify-center text-[10px] font-bold shadow-sm ${isOccupied ? 'occupied text-white' : (isSelected ? 'selected text-white' : 'bg-green-500 text-white')}`; seat.innerText = id; if (!isOccupied) { seat.onclick = () => { if (selectedSeats.includes(id)) { selectedSeats = selectedSeats.filter(s => s !== id); } else { selectedSeats.push(id); } // Update the bottom bar const bar = document.getElementById('purchase-bar'); if (selectedSeats.length > 0) { bar.classList.remove('hidden'); document.getElementById('selected-list-text').innerText = selectedSeats.join(', '); document.getElementById('bar-total').innerText = selectedSeats.length * 10; } else { bar.classList.add('hidden'); } renderGrid(taken); // Refresh colors }; } grid.appendChild(seat); } }); }
